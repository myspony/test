#include <iostream>

int main() {
    int num, factorial = 1;
    std::cout << "Enter a positive number : ";
    std::cin >> num;
    int i = 1;
    while (i <= num) {
        factorial *= i;
        ++i;
    }
    std::cout << "Factorial of " << num << " is " << factorial << std::endl;
    return 0;
}
